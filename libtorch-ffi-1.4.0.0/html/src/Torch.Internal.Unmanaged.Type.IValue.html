class IValueLike a b where
  toIValue :: a -> IO b
  fromIValue :: b -> IO a

instance IValueLike (Ptr IValue) (Ptr IValue) where
  toIValue _x =
    [C.throwBlock| at::IValue* { return new at::IValue(
      *$(at::IValue* _x));
    }|]
  fromIValue _obj =
    [C.throwBlock| at::IValue* { return new at::IValue((*$(at::IValue* _obj)).toIValue(
      ));
    }|]
